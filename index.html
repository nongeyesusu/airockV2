import cv2
import numpy as np
import mss
from ultralytics import YOLO
import time
import pydirectinput

# ------------------------------
# Settings (‡∏ï‡∏±‡πâ‡∏á‡∏Ñ‡πà‡∏≤‡∏û‡∏∑‡πâ‡∏ô‡∏ê‡∏≤‡∏ô)
# ------------------------------
MODEL_PATH = 'rock_miner_v1.pt'  # ‡πÑ‡∏ü‡∏•‡πå‡∏™‡∏°‡∏≠‡∏á AI
CONFIDENCE_THRESHOLD = 0.8       # ‡∏Ñ‡∏ß‡∏≤‡∏°‡∏°‡∏±‡πà‡∏ô‡πÉ‡∏à 80% ‡∏Ç‡∏∂‡πâ‡∏ô‡πÑ‡∏õ‡∏ñ‡∏∂‡∏á‡∏à‡∏∞‡∏ô‡∏±‡∏ö‡∏ß‡πà‡∏≤‡πÄ‡∏õ‡πá‡∏ô‡∏´‡∏¥‡∏ô
SCREEN_WIDTH = 1920
SCREEN_HEIGHT = 1080

CENTER_X = SCREEN_WIDTH // 2     # ‡∏à‡∏∏‡∏î‡∏Å‡∏∂‡πà‡∏á‡∏Å‡∏•‡∏≤‡∏á‡∏à‡∏≠ (‡∏ï‡∏±‡∏ß‡∏ú‡∏π‡πâ‡πÄ‡∏•‡πà‡∏ô)
CENTER_Y = SCREEN_HEIGHT // 2

YOLO_INPUT_W = 960               # ‡∏¢‡πà‡∏≠‡∏†‡∏≤‡∏û‡πÄ‡∏û‡∏∑‡πà‡∏≠‡πÉ‡∏´‡πâ AI ‡∏õ‡∏£‡∏∞‡∏°‡∏ß‡∏•‡∏ú‡∏•‡πÄ‡∏£‡πá‡∏ß‡∏Ç‡∏∂‡πâ‡∏ô
YOLO_INPUT_H = 540
FRAME_SKIP = 2                   # ‡∏Ç‡πâ‡∏≤‡∏°‡πÄ‡∏ü‡∏£‡∏° ‡πÄ‡∏û‡∏∑‡πà‡∏≠‡∏õ‡∏£‡∏∞‡∏´‡∏¢‡∏±‡∏î CPU

SMOOTHING_FACTOR = 0.45
smoothed_mn = None

pydirectinput.PAUSE = 0.01       # ‡∏•‡∏î‡∏î‡∏µ‡πÄ‡∏•‡∏¢‡πå‡∏Å‡∏≤‡∏£‡∏Å‡∏î‡∏õ‡∏∏‡πà‡∏°

# Load model
model = YOLO(MODEL_PATH)

# =====================================
# DRAW + LOGIC (‡∏´‡∏±‡∏ß‡πÉ‡∏à‡∏´‡∏•‡∏±‡∏Å)
# =====================================
def draw_boxes_and_lines(frame, results):
    global smoothed_mn
    
    start_point = (CENTER_X, CENTER_Y)
    stones = []
    
    # ------------------------------
    # ‡πÄ‡∏Å‡πá‡∏ö bbox (‡πÄ‡∏â‡∏û‡∏≤‡∏∞ ROCK)
    # ------------------------------
    for r in results:
        boxes = r.boxes
        if boxes is None:
            continue
        
        for box in boxes:
            cls = int(box.cls[0])
            name = model.names[cls]
            
            if name != "rock":
                continue
            
            bx1, by1, bx2, by2 = box.xyxy[0].cpu().numpy()
            
            # scale ‡πÄ‡∏û‡∏¥‡πà‡∏°‡∏Ñ‡∏ß‡∏≤‡∏°‡∏•‡∏∑‡πà‡∏ô ‡πÑ‡∏°‡πà‡πÅ‡∏Å‡πâ tensor
            scale_x = SCREEN_WIDTH / YOLO_INPUT_W
            scale_y = SCREEN_HEIGHT / YOLO_INPUT_H
            
            x1 = int(bx1 * scale_x)
            y1 = int(by1 * scale_y)
            x2 = int(bx2 * scale_x)
            y2 = int(by2 * scale_y)
            
            stones.append((x1, y1, x2, y2))
            
            # ‡∏ß‡∏≤‡∏î‡∏Å‡∏£‡∏≠‡∏ö‡∏´‡∏¥‡∏ô
            cv2.rectangle(frame, (x1, y1), (x2, y2), (0, 255, 0), 2)
    
    # ------------------------------
    # ‡πÑ‡∏°‡πà‡∏°‡∏µ‡∏´‡∏¥‡∏ô ‚Üí return ‡πÑ‡∏ß
    # ------------------------------
    if not stones:
        return frame
    
    # ------------------------------
    # ‡∏ß‡∏≤‡∏î‡∏à‡∏∏‡∏î player ‡∏Å‡∏•‡∏≤‡∏á‡∏à‡∏≠
    # ------------------------------
    cv2.circle(frame, start_point, 6, (255, 0, 0), -1)
    
    # ------------------------------
    # ‡∏´‡∏≤ rock ‡πÉ‡∏Å‡∏•‡πâ‡∏ó‡∏µ‡πà‡∏™‡∏∏‡∏î
    # ------------------------------
    mn = float('inf')
    nearest = None
    
    for (x1, y1, x2, y2) in stones:
        px, py = start_point
        
        dx = max(x1 - px, 0, px - x2)
        dy = max(y1 - py, 0, py - y2)
        dist = np.sqrt(dx * dx + dy * dy)
        
        # ‡∏à‡∏∏‡∏î‡πÉ‡∏Å‡∏•‡πâ‡∏™‡∏∏‡∏î‡∏Ç‡∏≠‡∏á‡∏´‡∏¥‡∏ô
        cx = min(max(px, x1), x2)
        cy = min(max(py, y1), y2)
        
        cv2.line(frame, start_point, (cx, cy), (255, 255, 0), 2)
        
        cv2.putText(frame, f"{dist:.1f}", (cx + 5, cy),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.6, (255,255,0), 2)
        
        if dist < mn:
            mn = dist
            nearest = ( (x1+x2)//2, (y1+y2)//2 )
    
    # ------------------------------
    # SMOOTHING ‡∏£‡∏∞‡∏¢‡∏∞
    # ------------------------------
    if smoothed_mn is None:
        smoothed_mn = mn
    else:
        smoothed_mn = smoothed_mn*0.6 + mn*0.4
    
    # ‡∏ß‡∏≤‡∏î‡∏ß‡∏á‡∏Å‡∏•‡∏°‡∏´‡∏¥‡∏ô‡∏ó‡∏µ‡πà‡πÉ‡∏Å‡∏•‡πâ‡∏™‡∏∏‡∏î
    cv2.circle(frame, nearest, 8, (0, 255, 255), -1)
    
    # ------------------------------
    # ACTION ‡∏•‡∏∑‡πà‡∏ô‡πÜ
    # ------------------------------
    px, py = start_point
    tx, ty = nearest
    
    # ----------------------
    # ‡∏ñ‡πâ‡∏≤‡πÉ‡∏Å‡∏•‡πâ‡∏°‡∏≤‡∏Å ‚Üí click
    # ----------------------
    if mn < 80:
        pydirectinput.click()
        cv2.putText(frame, "MINE!", (tx, ty - 20),
                    cv2.FONT_HERSHEY_SIMPLEX, 0.8, (0,255,0), 2)
        return frame
    
    # ----------------------
    # ‡πÄ‡∏î‡∏¥‡∏ô‡πÄ‡∏Ç‡πâ‡∏≤‡∏´‡∏≤‡∏´‡∏¥‡∏ô
    # ----------------------
    dx = tx - px
    dy = ty - py
    
    if abs(dx) > abs(dy):
        key = "d" if dx > 0 else "a"
    else:
        key = "s" if dy > 0 else "w"
    
    # non-blocking walk
    pydirectinput.keyDown(key)
    time.sleep(0.05)
    pydirectinput.keyUp(key)
    
    cv2.putText(frame, f"WALK {key.upper()}", (px, py - 40),
                cv2.FONT_HERSHEY_SIMPLEX, 1, (0,0,255), 2)
    
    return frame

# =====================================
# MAIN LOOP (optimized)
# =====================================
def capture_until_detected():
    frame_id = 0
    last_res = None
    
    cv2.namedWindow("YOLO Detection", cv2.WINDOW_NORMAL)
    cv2.resizeWindow("YOLO Detection", 960, 540)
    
    with mss.mss() as sct:
        monitor = {"top": 0, "left": 0, "width": SCREEN_WIDTH, "height": SCREEN_HEIGHT}
        
        while True:
            frame_id += 1
            
            img = np.array(sct.grab(monitor))
            frame = cv2.cvtColor(img, cv2.COLOR_BGRA2BGR)
            
            # YOLO ‡∏ó‡∏∏‡∏Å N ‡πÄ‡∏ü‡∏£‡∏°
            if frame_id % FRAME_SKIP == 0:
                small = cv2.resize(frame, (YOLO_INPUT_W, YOLO_INPUT_H))
                last_res = model.predict(small, conf=CONFIDENCE_THRESHOLD, verbose=False)
            
            if last_res:
                frame = draw_boxes_and_lines(frame, last_res)
            
            cv2.imshow("YOLO Detection", frame)
            
            if cv2.waitKey(1) & 0xFF == ord("q"):
                break
        
        cv2.destroyAllWindows()

# Run
capture_until_detected()
